Model regresji liniowej:

m1 <- lm(I(log(price)) ~ I(carat^(1/3)), data = trainset)

m2 <- update(m1, ~ . + carat)
m3 <- update(m2, ~ . + cut)
m4 <- update(m3, ~ . + color)
m5 <- update(m4, ~ . + clarity)
summary(m5)
